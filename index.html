import networkx as nx
import matplotlib.pyplot as plt
import matplotlib.lines as mlines
import random

def watts_strogatz_graph(n, k, p):
    """
    Generates a Watts-Strogatz small-world graph.

    Args:
        n (int): The number of nodes.
        k (int): Each node is joined with its `k` nearest neighbors 
                 in a ring topology. `k` must be an even integer.
        p (float): The probability of rewiring each edge.

    Returns:
        networkx.Graph: The generated small-world graph.
    """
    if k % 2 != 0:
        raise ValueError("k must be an even integer.")
    if not 0 <= p <= 1:
        raise ValueError("p must be between 0 and 1.")

    # 1. START: Create a regular ring lattice
    G = nx.Graph()
    nodes = list(range(n))
    G.add_nodes_from(nodes)

    # Connect each node to its k nearest neighbors
    for i in range(n):
        for j in range(1, k // 2 + 1):
            neighbor = (i + j) % n
            G.add_edge(i, neighbor)
            
    initial_edges = set(G.edges())

    # 2. EVOLVE: Rewire edges with probability p
    for u, v in list(initial_edges):
        if random.random() < p:
            # Choose a new node to connect to
            possible_new_nodes = set(nodes) - {u} - set(nx.neighbors(G, u))
            if not possible_new_nodes:
                continue

            new_v = random.choice(list(possible_new_nodes))
            
            # Rewire the edge
            G.remove_edge(u, v)
            G.add_edge(u, new_v)

    return G, initial_edges
# --- Parameters for the model ---
N = 20  # Number of nodes
K = 4   # Number of nearest neighbors for the initial ring
P_small = 0.2 # Small rewiring probability for the "sweet spot"

# --- Generate all three graphs ---
print("Generating graphs for the demonstration...")
regular_graph, _ = watts_strogatz_graph(N, K, 0)
small_world_graph, initial_edges = watts_strogatz_graph(N, K, P_small)
random_graph, _ = watts_strogatz_graph(N, K, 1)

# --- Visualizations ---
pos = nx.circular_layout(regular_graph)

# 1. Regular Network (p=0)
plt.figure(figsize=(8, 8))
plt.title("1. Regular Network (p=0)", fontsize=16)
nx.draw(regular_graph, pos, with_labels=True, 
        node_color='skyblue', node_size=700, edge_color='gray')

# 2. Small-World Graph (The "Sweet Spot")
plt.figure(figsize=(8, 8))
plt.title(f"2. The 'Sweet Spot' (p={P_small})", fontsize=16)

# Identify original vs rewired edges
original_edges = [edge for edge in small_world_graph.edges() 
                  if edge in initial_edges or (edge[1], edge[0]) in initial_edges]
rewired_edges = [edge for edge in small_world_graph.edges() 
                 if edge not in original_edges and (edge[1], edge[0]) not in original_edges]

About This Project

The Watts-Strogatz Model

The small-world network model, introduced by Duncan Watts and Steven Strogatz in 1998, explains how real-world networks can have both high clustering (like regular networks) and short average path lengths (like random networks).

This phenomenon explains the "six degrees of separation" concept and is observed in social networks, neural networks, power grids, and the internet.

Implementation Details

  • Built with Python 3.13, NetworkX, and Matplotlib
  • Interactive visualizations showing network evolution
  • Configurable parameters (N, K, p) for experimentation
  • Clear distinction between original and rewired edges

Dataset Characteristics

This dataset represents a real-world email communication network from a large European research institution. The network is directed, meaning email exchanges have a specific sender and receiver. With over 1,000 members and 25,000+ communications, this network exhibits typical characteristics of organizational communication patterns:

  • Scale-free properties: A few highly connected hubs with many peripheral nodes
  • Small-world characteristics: Short average path lengths despite low overall density
  • Community structure: Natural clustering based on research groups and departments
  • Hierarchical organization: Varying levels of influence and centrality across members

Link Analysis & Centrality

Identifying influential nodes using PageRank and Eigenvector Centrality algorithms

Understanding Network Centrality

Link analysis reveals which members play crucial roles in the communication network. We employ two complementary centrality measures:

PageRank Centrality

Originated from Google's web search algorithm, PageRank identifies influential nodes based on both the quantity and quality of incoming connections. A node is important if it receives emails from other important nodes.

Eigenvector Centrality

Measures the influence of a node based on the influence of its neighbors. Nodes connected to highly-connected neighbors score higher, revealing members embedded in influential network regions.

Community Detection

Discovering natural clusters in the email network using Label Propagation

Label Propagation Algorithm

Community detection reveals the underlying organizational structure of the email network. We use the asynchronous Label Propagation algorithm, which identifies groups of nodes that communicate more frequently with each other than with the rest of the network.

Algorithm Process

  1. Each node is initialized with a unique label
  2. Nodes iteratively adopt the most common label among their neighbors
  3. The process continues until labels stabilize
  4. Nodes with the same final label form a community

Result: The algorithm identified distinct communities corresponding to research departments, project teams, and administrative units within the institution.

Influence Analysis

Combining PageRank and Betweenness Centrality to quantify member influence

Multi-Metric Influence Scoring

True influence in a network comes from multiple dimensions. We combine two complementary centrality measures to create a comprehensive influence score:

PageRank Influence

What it measures: Prestige and authority within the network

Interpretation: High PageRank indicates receiving communications from other influential members

Betweenness Centrality

What it measures: Information brokerage and control

Interpretation: High betweenness indicates serving as a bridge between different network regions

Combined Influence Score = 0.5 √ó (Normalized PageRank) + 0.5 √ó (Normalized Betweenness)

This balanced approach identifies both prestigious members and critical connectors in the communication network.

Anomaly Detection

Identifying unusual communication patterns using Isolation Forest

Detecting Anomalous Behavior

Anomaly detection reveals members with unusual communication patterns that deviate significantly from normal network behavior. These outliers may represent:

  • Automated systems: Bots or mailing lists with abnormal sending patterns
  • External collaborators: Individuals with limited internal connections
  • Isolated specialists: Experts with unique communication profiles
  • Data quality issues: Potential errors in data collection

Isolation Forest Algorithm

We use Isolation Forest, an unsupervised machine learning algorithm that identifies anomalies by measuring how easily observations can be isolated from the rest of the data.

Features Used: Degree centrality, betweenness centrality, closeness centrality, clustering coefficient, and PageRank

Contamination Rate: 5% (assuming ~50 anomalous nodes in the network)

Link Prediction

Forecasting future email connections using graph-based features and machine learning

Predicting Future Connections

Link prediction addresses the question: "Which members are likely to start communicating who haven't yet?" This has practical applications in:

  • Collaboration recommendations: Suggesting potential research partners
  • Network growth modeling: Understanding how communication networks evolve
  • Resource allocation: Anticipating future communication infrastructure needs
  • Team formation: Identifying promising cross-departmental collaborations

Prediction Methodology

Features Extracted: For each potential node pair (u, v), we compute:

  • Common Neighbors: Number of shared connections
  • Jaccard Coefficient: Normalized measure of neighborhood overlap
  • Adamic-Adar Index: Weighted common neighbors emphasizing rare connections
  • Preferential Attachment: Product of node degrees (rich-get-richer effect)
  • Resource Allocation: Flow-based similarity measure

Classifier: Random Forest with 100 estimators trained on 80% of data, tested on 20%

Model Performance

ROC Curve Analysis

The Receiver Operating Characteristic (ROC) curve evaluates the model's ability to distinguish between true future connections and non-connections across all classification thresholds. The Area Under Curve (AUC) quantifies overall model performance, with values closer to 1.0 indicating better prediction accuracy. Our model demonstrates strong predictive power, significantly outperforming random guessing (AUC = 0.5).

def task_link_analysis(G: nx.Graph, topk: int = 20):
    """Link Analysis: PageRank and Eigenvector centrality"""
    pr = nx.pagerank(G)
    ev = nx.eigenvector_centrality(G, max_iter=1000)
    
    df = pd.DataFrame({
        "node": list(pr.keys()),
        "pagerank": list(pr.values()),
        "eigenvector": [ev[n] for n in pr.keys()],
        "degree": [G.degree(n) for n in pr.keys()],
    }).sort_values("pagerank", ascending=False)
    
    return df

def task_node_classification(G: nx.Graph):
    """Community Detection via Label Propagation"""
    comms = list(nx.algorithms.community.asyn_lpa_communities(G))
    node2comm = {}
    for cid, nodes in enumerate(comms):
        for n in nodes:
            node2comm[n] = cid
    return node2comm

def task_influence(G: nx.Graph):
    """Influence Analysis: PageRank and Betweenness"""
    pr = nx.pagerank(G)
    bt = nx.betweenness_centrality(G, normalized=True)
    
    df = pd.DataFrame({
        "node": list(pr.keys()),
        "pagerank": list(pr.values()),
        "betweenness": [bt[n] for n in pr.keys()],
    }).sort_values(["pagerank", "betweenness"], ascending=False)
    
    return df

def task_anomaly_detection(G: nx.Graph, contamination=0.01):
    """Anomaly Detection using IsolationForest"""
    degrees = dict(G.degree())
    clustering = nx.clustering(G)
    avg_neigh_deg = nx.average_neighbor_degree(G)
    
    X = [[degrees[n], clustering[n], avg_neigh_deg[n]] 
         for n in G.nodes()]
    
    iso = IsolationForest(n_estimators=300, 
                         contamination=contamination)
    anomaly_scores = iso.fit_predict(X)
    
    return anomaly_scores
